```typescript
  async getRelatorioUso(
    tenantId: number,
    obraId: number,
    dataInicio: string,
    dataFim: string,
  ) {
    const obra = await this.prisma.obra.findFirst({
      where: { id: obraId, tenantId, deletadoEm: null },
      select: { nome: true },
    });
    if (!obra) throw new Error('Obra não encontrada');

    const fichas = await this.prisma.fvsFicha.findMany({
      where: {
        obraId,
        tenantId,
        deletadoEm: null,
        criadoEm: { gte: new Date(dataInicio), lte: new Date(dataFim) },
      },
      include: {
        servicos: {
          include: {
            servico: { select: { nome: true } },
            registros: {
              select: { status: true },
            },
          },
        },
        criadoPor: { select: { nome: true } },
      },
      orderBy: { criadoEm: 'asc' },
    });

    // Aggregate por serviço
    const porServico: Record<string, {
      servico_nome: string;
      total_fichas: number;
      total_registros: number;
      registros_ok: number;
      registros_nc: number;
      taxa: number;
    }> = {};

    for (const ficha of fichas) {
      for (const svc of ficha.servicos) {
        const nome = svc.servico.nome;
        if (!porServico[nome]) {
          porServico[nome] = {
            servico_nome: nome,
            total_fichas: 0,
            total_registros: 0,
            registros_ok: 0,
            registros_nc: 0,
            taxa: 0,
          };
        }
        porServico[nome].total_fichas += 1;
        for (const reg of svc.registros) {
          porServico[nome].total_registros += 1;
          if (reg.status === 'CONFORME') porServico[nome].registros_ok += 1;
          if (reg.status === 'NAO_CONFORME') porServico[nome].registros_nc += 1;
        }
      }
    }

    // Compute taxa
    for (const row of Object.values(porServico)) {
      row.taxa = row.total_registros > 0
        ? Math.round((row.registros_ok / row.total_registros) * 100)
        : 0;
    }

    // Aggregate por inspetor
    const porInspetor: Record<string, {
      inspetor_nome: string;
      total_fichas: number;
      fichas_concluidas: number;
    }> = {};

    for (const ficha of fichas) {
      const nome = ficha.criadoPor?.nome ?? 'Desconhecido';
      if (!porInspetor[nome]) {
        porInspetor[nome] = { inspetor_nome: nome, total_fichas: 0, fichas_concluidas: 0 };
      }
      porInspetor[nome].total_fichas += 1;
      if (ficha.status === 'concluida' || ficha.status === 'aprovada') {
        porInspetor[nome].fichas_concluidas += 1;
      }
    }

    return {
      obra_nome: obra.nome,
      data_inicio: dataInicio,
      data_fim: dataFim,
      total_fichas: fichas.length,
      por_servico: Object.values(porServico).sort((a, b) => b.total_fichas - a.total_fichas),
      por_inspetor: Object.values(porInspetor).sort((a, b) => b.total_fichas - a.total_fichas),
    };
  }
```

```typescript
// frontend-web/src/modules/fvs/relatorios/excel/UsoXlsx.ts
import ExcelJS from 'exceljs';
import type { R6UsoData } from '../types';

const HEADER_FONT: Partial<ExcelJS.Font> = { bold: true, color: { argb: 'FFFFFFFF' }, name: 'Calibri', size: 10 };
const HEADER_FILL: ExcelJS.Fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FF2563EB' } };
const BORDER: Partial<ExcelJS.Borders> = {
  top:    { style: 'thin', color: { argb: 'FFE5E7EB' } },
  left:   { style: 'thin', color: { argb: 'FFE5E7EB' } },
  bottom: { style: 'thin', color: { argb: 'FFE5E7EB' } },
  right:  { style: 'thin', color: { argb: 'FFE5E7EB' } },
};

function applyHeader(row: ExcelJS.Row) {
  row.eachCell((c) => {
    c.font = HEADER_FONT; c.fill = HEADER_FILL; c.border = BORDER;
    c.alignment = { vertical: 'middle', horizontal: 'center' };
  });
  row.height = 22;
}

function applyData(row: ExcelJS.Row, alt: boolean) {
  row.eachCell((c) => {
    c.border = BORDER;
    c.alignment = { vertical: 'middle' };
    if (alt) c.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FFF9FAFB' } };
  });
}

export async function gerarUsoXlsx(dados: R6UsoData): Promise<ArrayBuffer> {
  const wb = new ExcelJS.Workbook();
  wb.creator = 'Eldox';
  wb.created = new Date();

  // Sheet 1: Por Serviço
  const ws1 = wb.addWorksheet('Por Serviço');
  ws1.columns = [
    { header: 'Serviço',          key: 'servico_nome',    width: 28 },
    { header: 'Fichas',           key: 'total_fichas',    width: 10 },
    { header: 'Registros',        key: 'total_registros', width: 12 },
    { header: 'Conformes',        key: 'registros_ok',    width: 12 },
    { header: 'Não Conformes',    key: 'registros_nc',    width: 14 },
    { header: 'Taxa (%)',         key: 'taxa',            width: 10 },
  ];
  applyHeader(ws1.getRow(1));
  ws1.views = [{ state: 'frozen', ySplit: 1 }];

  if (dados.por_servico.length === 0) {
    ws1.addRow({ servico_nome: 'Nenhum registro encontrado' }).getCell(1).font = { italic: true };
  } else {
    dados.por_servico.forEach((row, i) => {
      const r = ws1.addRow(row);
      applyData(r, i % 2 === 1);
      const taxaCell = r.getCell('taxa');
      if (row.taxa >= 90) taxaCell.font = { bold: true, color: { argb: 'FF16A34A' } };
      else if (row.taxa >= 70) taxaCell.font = { bold: true, color: { argb: 'FFD97706' } };
      else taxaCell.font = { bold: true, color: { argb: 'FFDC2626' } };
    });
    // Total row
    const total = ws1.addRow({
      servico_nome: 'TOTAL',
      total_fichas: dados.total_fichas,
      total_registros: dados.por_servico.reduce((s, r) => s + r.total_registros, 0),
      registros_ok: dados.por_servico.reduce((s, r) => s + r.registros_ok, 0),
      registros_nc: dados.por_servico.reduce((s, r) => s + r.registros_nc, 0),
      taxa: '',
    });
    total.eachCell((c) => {
      c.font = { bold: true };
      c.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FFDBEAFE' } };
      c.border = BORDER;
    });
  }

  // Sheet 2: Por Inspetor
  const ws2 = wb.addWorksheet('Por Inspetor');
  ws2.columns = [
    { header: 'Inspetor',         key: 'inspetor_nome',      width: 28 },
    { header: 'Fichas',           key: 'total_fichas',        width: 10 },
    { header: 'Concluídas',       key: 'fichas_concluidas',   width: 12 },
    { header: '% Conclusão',      key: 'pct',                 width: 14 },
  ];
  applyHeader(ws2.getRow(1));
  ws2.views = [{ state: 'frozen', ySplit: 1 }];

  dados.por_inspetor.forEach((row, i) => {
    const pct = row.total_fichas > 0
      ? Math.round((row.fichas_concluidas / row.total_fichas) * 100)
      : 0;
    const r = ws2.addRow({ ...row, pct });
    applyData(r, i % 2 === 1);
  });

  // Sheet 3: Cabeçalho / Resumo
  const ws3 = wb.addWorksheet('Resumo');
  ws3.columns = [
    { header: 'Campo',  key: 'campo',  width: 22 },
    { header: 'Valor',  key: 'valor',  width: 30 },
  ];
  applyHeader(ws3.getRow(1));
  [
    { campo: 'Obra',             valor: dados.obra_nome },
    { campo: 'Período de',       valor: new Date(dados.data_inicio).toLocaleDateString('pt-BR') },
    { campo: 'Período até',      valor: new Date(dados.data_fim).toLocaleDateString('pt-BR') },
    { campo: 'Total de Fichas',  valor: dados.total_fichas },
    { campo: 'Serviços inspecionados', valor: dados.por_servico.length },
  ].forEach((item, i) => {
    const r = ws3.addRow(item);
    applyData(r, i % 2 === 1);
  });

  return wb.xlsx.writeBuffer() as Promise<ArrayBuffer>;
}
```

  async duplicarLocal(tenantId: number, obraId: number, localId: number) {
    const original = await this.prisma.obraLocal.findFirst({
      where: { id: localId, obraId, tenantId, deletadoEm: null },
    });
    if (!original) throw new NotFoundException('Local não encontrado');

    // Find max ordem among siblings
    const irmao = await this.prisma.obraLocal.findFirst({
      where: {
        obraId,
        tenantId,
        parentId: original.parentId ?? null,
        deletadoEm: null,
      },
      orderBy: { ordem: 'desc' },
    });
    const novaOrdem = (irmao?.ordem ?? 0) + 1;

    // Generate new codigo
    const novoCodigo = `${original.codigo}-copia-${Date.now()}`.slice(0, 30);

    const copia = await this.prisma.obraLocal.create({
      data: {
        tenantId,
        obraId,
        parentId: original.parentId ?? null,
        nivel: original.nivel,
        nome: `${original.nome} (cópia)`,
        codigo: novoCodigo,
        nomeCompleto: `${original.nomeCompleto} (cópia)`,
        status: 'PENDENTE',
        ordem: novaOrdem,
      },
    });

    // Recursively duplicate children
    await this.duplicarFilhos(tenantId, obraId, localId, copia.id);

    return copia;
  }

  private async duplicarFilhos(
    tenantId: number,
    obraId: number,
    parentOriginalId: number,
    parentNovoId: number,
  ) {
    const filhos = await this.prisma.obraLocal.findMany({
      where: { parentId: parentOriginalId, obraId, tenantId, deletadoEm: null },
      orderBy: { ordem: 'asc' },
    });
    for (const filho of filhos) {
      const copiaFilho = await this.prisma.obraLocal.create({
        data: {
          tenantId,
          obraId,
          parentId: parentNovoId,
          nivel: filho.nivel,
          nome: filho.nome,
          codigo: `${filho.codigo}-cp-${Date.now()}`.slice(0, 30),
          nomeCompleto: filho.nomeCompleto,
          status: 'PENDENTE',
          ordem: filho.ordem,
        },
      });
      await this.duplicarFilhos(tenantId, obraId, filho.id, copiaFilho.id);
    }
  }

### Key invariants

- The `bloqueado` status string is lowercase — matches the backend's free-form `status: string` field in Prisma. When unblocking, we restore `'PENDENTE'` (uppercase) to match existing status values in `STATUS_LOCAL_LABEL`.
- `podeDuplicar={nivelAtual === 1}` means only top-level items (torres) show the duplicate button. Adapts automatically if the obra type has different hierarchy names.
- The `duplicarFilhos` recursion uses `Date.now()` in the generated codigo to avoid unique constraint collisions when multiple children are copied in the same millisecond. This is fine for a code that is used only internally.
- `R6_USO` only exposes `excel` format in the dashboard button because the PDF renderer has no R6 case. The `FiltrosModal` supports it but only excel is wired in the shortcut. Adding a PDF renderer is a separate task if ever needed.
